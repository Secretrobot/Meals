<script>


/* =========================

   STATE MANAGER

========================= */


const State = {

  data:{

    week: Array(7).fill(null).map(()=>({meal:null,locked:false})),

    usage:{},

    likes:{},

    avoids:{},

    filters:[]

  },


  async load(){

    const saved = await Storage.get("state")

    if(saved) this.data = saved

  },


  async save(){

    await Storage.set("state", this.data)

  }

}


/* =========================

   STORAGE (IndexedDB)

========================= */


const Storage = (()=>{


  const DB_NAME="meal_planner_v5"

  const STORE="kv"

  const VERSION=1


  let db


  function open(){

    return new Promise((res)=>{

      const req=indexedDB.open(DB_NAME,VERSION)


      req.onupgradeneeded=e=>{

        const db=e.target.result

        db.createObjectStore(STORE)

      }


      req.onsuccess=e=>{

        db=e.target.result

        res()

      }

    })

  }


  async function get(key){

    if(!db) await open()

    return new Promise(r=>{

      const tx=db.transaction(STORE,"readonly")

      const store=tx.objectStore(STORE)

      const req=store.get(key)

      req.onsuccess=()=>r(req.result)

    })

  }


  async function set(key,val){

    if(!db) await open()

    return new Promise(r=>{

      const tx=db.transaction(STORE,"readwrite")

      tx.objectStore(STORE).put(val,key)

      tx.oncomplete=()=>r()

    })

  }


  return {open,get,set}

})()


/* =========================

   MEAL DATA

========================= */


const meals = [

"Fried Rice","Noodle Bowl","Quesadilla","Pesto Pasta","Thai Curry",

"Halloumi Wraps","Black Bean Rice","Stir Fry","Baked Potato",

"Sunday Roast","Veggie Burgers","Grilled Cheese + Soup",

"Chickpea Curry","Aloo Gobi","Falafel and Veg",

"Shepherd's Pie","Sweet Potato Black Bean Tacos"

]


/* =========================

   RECOMMENDATION ENGINE

========================= */


const Engine = {


  score(meal, dayIndex){


    const s = State.data


    let score = 0


    const freq = s.usage[meal] || 0

    const liked = s.likes[meal] || 0

    const avoided = s.avoids[meal] || 0


    // frequency decay

    score -= freq * 2


    // preference learning

    score += liked * 3

    score -= avoided * 6


    // weekly repetition penalty

    if(s.week.some(d=>d.meal===meal)) score -= 8


    // weekend bias

    if(dayIndex>=5){

      if(meal.includes("Roast") || meal.includes("Pasta")) score += 2

    }


    // quick/comfort heuristics

    if(meal.includes("Burger") || meal.includes("Wrap")) score += 1

    if(meal.includes("Cheese") || meal.includes("Curry")) score += 1


    return score

  },


  pick(dayIndex){


    const ranked = meals

      .map(m=>({m,score:this.score(m,dayIndex)}))

      .sort((a,b)=>b.score-a.score)


    const top = ranked.slice(0,5)

    return top[Math.floor(Math.random()*top.length)].m

  }

}


/* =========================

   UX ENGINE (learning signals)

========================= */


const Learn = {


  select(meal){

    const s = State.data

    s.likes[meal] = (s.likes[meal]||0) + 1

    s.usage[meal] = (s.usage[meal]||0) + 1

  },


  reroll(meal){

    const s = State.data

    s.avoids[meal] = (s.avoids[meal]||0) + 1

  }

}


/* =========================

   UI

========================= */


let activeDay=null


function render(){

  const grid=document.getElementById("grid")

  grid.innerHTML=""


  State.data.week.forEach((d,i)=>{


    const el=document.createElement("div")

    el.className="day" + (d.locked?" locked":"")

    el.onclick=()=>openModal(i)


    el.innerHTML=`

      <h4>Day ${i+1}</h4>

      <div class="meal">${d.meal||""}</div>


      <div class="icons">

        <button onclick="reroll(${i});event.stopPropagation()">↻</button>

        <button onclick="toggleLock(${i});event.stopPropagation()">🔒</button>

      </div>

    `


    grid.appendChild(el)

  })

}


function openModal(i){

  activeDay=i


  const modal=document.getElementById("modal")

  const content=document.getElementById("modalContent")


  const suggestions = getSuggestions(i)


  content.innerHTML=`

    <div class="chips">

      ${["Comfort","Quick","Italian","Asian","Mexican"].map(f=>`

        <div class="chip ${State.data.filters.includes(f)?"active":""}"

             onclick="toggleFilter('${f}')">

          ${f}

        </div>

      `).join("")}

    </div>


    ${suggestions.map(m=>`

      <div class="suggestion" onclick="selectMeal('${m}')">

        ${m}

      </div>

    `).join("")}


    <div class="footer-actions">

      <button class="small-btn" onclick="clearDay()">Clear</button>

      <button class="small-btn" onclick="closeModal()">Done</button>

    </div>

  `


  modal.classList.add("show")

}


function getSuggestions(i){


  let pool = meals


  const filters = State.data.filters

  if(filters.length){

    pool = pool.filter(m=>filters.some(f=>m.includes(f)))

  }


  return pool

    .map(m=>({m,score:Engine.score(m,i)}))

    .sort((a,b)=>b.score-a.score)

    .slice(0,6)

    .map(x=>x.m)

}


/* =========================

   ACTIONS

========================= */


async function autoPopulate(){


  State.data.week.forEach((d,i)=>{

    if(!d.locked && !d.meal){

      d.meal = Engine.pick(i)

      Learn.select(d.meal)

    }

  })


  await State.save()

  render()

}


async function selectMeal(m){

  State.data.week[activeDay].meal=m

  Learn.select(m)

  await State.save()

  closeModal()

  render()

}


async function reroll(i){

  const old = State.data.week[i].meal

  if(old) Learn.reroll(old)


  State.data.week[i].meal = Engine.pick(i)

  await State.save()

  render()

}


async function clearDay(){

  State.data.week[activeDay]={meal:null,locked:false}

  await State.save()

  closeModal()

  render()

}


function toggleLock(i){

  State.data.week[i].locked=!State.data.week[i].locked

  State.save()

  render()

}


function toggleFilter(f){

  const s=State.data

  s.filters = s.filters.includes(f)

    ? s.filters.filter(x=>x!==f)

    : [...s.filters,f]


  openModal(activeDay)

}


/* =========================

   MODAL

========================= */


function closeModal(){

  document.getElementById("modal").classList.remove("show")

}


/* =========================

   PWA BASICS

========================= */


function registerPWA(){


  const sw = `

  self.addEventListener('install',e=>{

    self.skipWaiting()

  })


  self.addEventListener('fetch',()=>{})

  `


  const blob = new Blob([sw],{type:'text/javascript'})

  const url = URL.createObjectURL(blob)


  navigator.serviceWorker?.register(url)

}


/* =========================

   EXPORT / IMPORT

========================= */


function exportData(){

  const blob = new Blob(

    [JSON.stringify(State.data)],

    {type:"application/json"}

  )


  const a=document.createElement("a")

  a.href=URL.createObjectURL(blob)

  a.download="meal-plan-backup.json"

  a.click()

}


/* =========================

   INIT

========================= */


(async function init(){

  await Storage.open()

  await State.load()

  render()

  registerPWA()

})()
